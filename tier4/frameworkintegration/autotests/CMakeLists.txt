    add_executable(${_testname} ${_testname}.cpp ${ARGN})

    ecm_mark_as_test(${_testname})
    target_link_libraries(${_testname} Qt5::Test Qt5::DBus KF5::KConfigWidgets KF5::ConfigCore KF5::KIconThemes KF5::KStyle KF5::KIOFileWidgets KF5::I18n)
endmacro()
